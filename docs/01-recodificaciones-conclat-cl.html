# Escala numérica: conserva el rango original (con sus etiquetas de valor) y
# manda NS/NR (y otros missing) a NA. A diferencia de usar zap_labels() (que
# pela TODAS las etiquetas), acá se modifica el vector labelled directamente:
# solo se reemplazan por NA los códigos de "missing", y solo se quitan del
# atributo `labels` esas mismas entradas, para que no aparezcan como filas
# fantasma con 0 casos en frq().
rec_escala <- function(x, missing = c(-888, -999)) {
  x_num <- as.numeric(haven::zap_labels(x))
  vals <- as.numeric(haven::zap_labels(x))
  x[vals %in% missing] <- NA
  labs <- attr(x, "labels")
  if (!is.null(labs)) attr(x, "labels") <- labs[!(labs %in% missing)]
  x
}
